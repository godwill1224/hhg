Add New User

// userRoutes.js const express = require('express'); const router = express.Router(); const upload = require('./upload'); const User = require('../models/userModel'); // Route to update profile image router.post('/update-profile-image', upload.single('profileImage'), async (req, res) => { try { if (!req.file) { return res.status(400).send('No file uploaded.'); } const imageUrl = `/uploads/${req.file.filename}`; const userId = req.body.userId; // Assume userId is sent in the request body const user = await User.findByIdAndUpdate( userId, { profileImage: imageUrl }, { new: true } ); if (!user) { return res.status(404).send('User not found.'); } res.status(200).json({ message: 'Profile image updated successfully', user }); } catch (error) { res.status(500).send('Error updating profile image.'); } }); module.exports = router;
